import sys,bisect
input=sys.stdin.readline
mod=10**9+7
N=2*(10**5)+1
def inv(x):
	return pow(x,mod-2,mod)
def nCr(n,r):
	return (f[n]*inv((f[r]*f[n-r])%mod))%mod
f=[1]*N
for i in range(2,N):
	f[i]=(f[i-1]*i)%mod
for _ in range(int(input())):
	n,m,k=map(int,input().split())
	a=sorted(list(map(int,input().split())))
	ans=0
	for i in range(n):
		r=bisect.bisect_right(a,a[i]+k)
		if r-i-1>=m-1:
			ans=(ans+nCr(r-i-1,m-1))%mod
	print(ans)
